import java.util.*;

class MapDemo {
    
    public static void main(String[] args) {
        
         HashMap<String,String>  map = new HashMap<>(); // unordered and unsorted by keys
        
        // LinkedHashMap<String,String> map = new LinkedHashMap<>(); // maintains insertion order
        
        // TreeMap<String,String> map = new TreeMap<>(); // sorted by keys
        
        map.put("India", "New Delhi");
        map.put("Japan", "Tokyo");
        map.put("China", "Beijing");
        map.put("Russia", "Moscow");
        // map.put("China", "abc"); 
        
        System.out.println(map); // toString
        
        //get value from keys
        String inputKey = "Russia";
        
        String value = map.get(inputKey);  //return value if it is existing otherwise returns null
        
        if(value != null) {
            System.out.println("Capital: "+value);
        }
        else {
           System.out.println("Country not found"); 
        }
        
        //get all keys
        Set<String> set = map.keySet();
        
        System.out.println("All Keys In Map");
        System.out.println(set);
        
        //get all values
        System.out.println("Values in map");
        for(String s : set) {
           String v =  map.get(s);
           System.out.println(v);
        }
        
        String myKey = "China";
        
        if(map.containsKey(myKey)) {
            
            map.put(myKey, "New China Captial");
        }
        else {
            System.out.println("not found");
        }
        
        System.out.println(map);
        
    }
}
