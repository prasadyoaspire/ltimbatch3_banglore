import java.util.*;

class Hall {
    private String hallName;
    private String contactNumber;
    private double costPerDay;
    private String ownerName;

    Hall(String hallName,String contactNumber,double costPerDay,String ownerName) {
        this.hallName = hallName;
        this.contactNumber = contactNumber;
        this.costPerDay = costPerDay;
        this.ownerName = ownerName;
    }
    
    public double getCostPerDay() {
        return costPerDay;
    }
    
    public String getHallName() {
        return hallName;
    }
   
    @Override
    public String toString() {
        return hallName+" "+contactNumber+" "+costPerDay+" "+ownerName;
    }
}

class HallPriceComparator implements Comparator<Hall> {
    
    @Override
    public int compare(Hall h1, Hall h2) {
        return  Double.compare(h1.getCostPerDay(),h2.getCostPerDay());
    }
}

class HallNameComparator implements Comparator<Hall> {
    
    @Override
    public int compare(Hall h1, Hall h2) {
        return  h1.getHallName().compareTo(h2.getHallName());
    }
}

class Main {
    
    public static void main(String[] args) {
        
        ArrayList<Hall> hallList = new ArrayList<>();
        
        hallList.add(new Hall("JME", "9632587410", 50000, "Kiran"));
        hallList.add(new Hall("GuestLine", "9632587410", 40000, "Kiran"));
        hallList.add(new Hall("DomTree", "9632587410", 30000, "Kiran"));
        hallList.add(new Hall("7Apple", "9632587410", 150000, "Rathod"));
        
        System.out.println("sort by cost per day");
        
        Collections.sort(hallList, new HallPriceComparator() );
        
        for(Hall h: hallList) {
            System.out.println(h);
        }
        
        System.out.println("sort by hall name");


        Collections.sort(hallList, new HallNameComparator() );
        
        for(Hall h: hallList) {
            System.out.println(h);
        }
    }
}
